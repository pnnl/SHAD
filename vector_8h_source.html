570  void _reserve(size_type n) {
571  size_type currentCapacity = Capacity();
572  if (currentCapacity >= n) return;
573 
574  rt::Locality insertLocality(0);
575  size_t lastBlock(0);
576  std::tie(insertLocality, lastBlock, std::ignore) =
577  _targetFromPosition(currentCapacity, kBlockSize);
578  size_t newLastBlock = n / kBlockSize;
579 
580  size_type blocksToAllocate = 1;
581  blocksToAllocate += newLastBlock - lastBlock;
582 
583  std::vector<size_type> newBlocks(rt::numLocalities(), 0);
584  for (size_t i = static_cast<uint32_t>(insertLocality), j = blocksToAllocate;
585  j > 0; ++i, --j) {
586  newBlocks[i % rt::numLocalities()]++;
587  }

591  if (newBlocks[i] == 0) continue;

1108  rt::Handle &handle, Vector<T, Allocator>::size_type position,
1109  IteratorType begin, IteratorType end) {

1111  size_type newSize(0);
1112 
1113  rt::executeAtWithRet(
1114  mainLocality_,
1115  [](const std::tuple<ObjectID, size_type, size_type> &args,
1116  size_type *size) {
1117  auto This = Vector<T, Allocator>::GetPtr(std::get<0>(args));
1118  std::lock_guard<rt::Lock> _(This->sizeCapacityLock_);
1119  auto position = std::get<1>(args);
1120  auto newElements = std::get<2>(args);
1121 
1122  if (position > This->size_ || position + newElements <= This->size_) {
1123  *size = This->size_;
1124  return;
1125  }
1126 
1127  size_type end = position + newElements;
1128  size_type growth = This->size_ > end ? 0 : end - This->size_;
1129 
1130  This->size_ += growth;
1131  if (This->size_ > This->capacity_) This->_reserve(This->size_);
1132 
1133  *size = This->size_;
1134  },
1135  std::make_tuple(oid_, position, newElements), &newSize);
1136 
1137  // Trying to insert in an invalid position.
1138  if (position > newSize) {
1139  throw std::out_of_range("AsyncInsertAt: position out of range");
1140  }
1141 
1142  size_type startingPoint = newSize - newElements;
1143 
1144  constexpr size_t kNumElements = 4000 / sizeof(value_type);
1145  static_assert(kNumElements >= 1,
1146  "We can't do this due to a series of unfortunate events");
1147 
1148  struct InsertMessage {
1149  ObjectID objID;
1150  size_type startPosition;
1151  size_type numElements;
1152  value_type elements[kNumElements];
1153 
1154  InsertMessage()
1155  : objID(ObjectID::kNullID), startPosition(0), numElements(0) {}
1156  };
1157 
1158  auto insertFunction = [](rt::Handle &, const InsertMessage &args) {
1159  auto This = Vector<T, Allocator>::GetPtr(args.objID);
1160  auto blockOffsetPair = This->_blockOffsetFromPosition(args.startPosition);
1161  size_type localBlock =
1162  This->_globlalBlockToLocalBlock(blockOffsetPair.first);
1163  std::copy(&args.elements[0], &args.elements[args.numElements],
1164  &This->dataBlocks_.at(localBlock)[blockOffsetPair.second]);
1165  };
1166 
1167  InsertMessage args;
1168  size_t spaceLeftInBlock = kBlockSize - (startingPoint % kBlockSize);
1169  // first block
1170  args.objID = oid_;
1171  args.startPosition = startingPoint;
1172  args.numElements =
1173  std::min(newElements, std::min(kNumElements, spaceLeftInBlock));
1174 
1175  std::copy(begin, begin + args.numElements, args.elements);
1176 
1177  rt::Locality target(0);
1178  size_t blockNumber(0);
1179  size_t offset(0);
1180  std::tie(target, blockNumber, offset) =
1181  _targetFromPosition(position, kBlockSize);
1182 
1183  rt::asyncExecuteAt(handle, target, insertFunction, args);
1184 
1185  begin += args.numElements;
1186  newElements -= args.numElements;
1187  startingPoint += args.numElements;
1188 
1189  // inner blocks
1190  while (begin != end && newElements > 0) {
1191  std::tie(target, blockNumber, offset) =
1192  _targetFromPosition(startingPoint, kBlockSize);
1193  args.startPosition = startingPoint;
1194 
1195  spaceLeftInBlock = kBlockSize - (startingPoint % kBlockSize);
1196  args.numElements =
1197  std::min(newElements, std::min(kNumElements, spaceLeftInBlock));
1198 
1199  std::copy(begin, begin + args.numElements, args.elements);
1200  rt::asyncExecuteAt(handle, target, insertFunction, args);
1201 
1202  begin += args.numElements;
1203  newElements -= args.numElements;
1204  startingPoint += args.numElements;
1205  }
