43 // apply_from is an utility function that performs compile-time unpacking of
44 // elements [i,N) from a N-ary tuple and invokes a callable on the unpacked
45 // elements.
46 //
48 template <size_t i, size_t N>
49 struct Apply {
50  template <typename F, typename T, typename... A>
51  static inline auto apply(F&& f, T&& t, A&&... a) {
52  return Apply<i, N - 1>::apply(::std::forward<F>(f), ::std::forward<T>(t),
53  ::std::get<N - 1>(::std::forward<T>(t)),
54  ::std::forward<A>(a)...);
55  }
56 };

58 template <size_t i>
59 struct Apply<i, i> {
60  template <typename F, typename T, typename... A>
61  static inline auto apply(F&& f, T&&, A&&... a) {
62  return ::std::forward<F>(f)(::std::forward<A>(a)...);
63  }
64 };

68  return Apply<i, ::std::tuple_size<::std::decay_t<T>>::value>::apply(
69  ::std::forward<F>(f), ::std::forward<T>(t));

89 template <typename ForwardIt, typename MapF, typename S, typename... Args>
90 S distributed_folding_map(ForwardIt first, ForwardIt last, MapF&& map_kernel,
91  const S& init_sol, Args&&... args) {
92  using itr_traits = distributed_iterator_traits<ForwardIt>;
93  auto localities = itr_traits::localities(first, last);
94  auto res = init_sol;
95  for (auto locality = localities.begin(), end = localities.end();
96  locality != end; ++locality) {
97  auto d_args = std::make_tuple(map_kernel, first, last, res, args...);
98  rt::executeAtWithRet(
99  locality,
100  [](const typeof(d_args)& d_args, S* result) {
101  *result = apply_from<1>(::std::get<0>(d_args),
102  ::std::forward<typeof(d_args)>(d_args));
103  },
104  d_args, &res);
105  }
106  return res;
107 }

189 template <typename ForwardIt, typename MapF, typename... Args>
190 std::vector<
191  typename std::result_of<MapF&(ForwardIt, ForwardIt, Args&&...)>::type>
192 distributed_map_init(
193  ForwardIt first, ForwardIt last, MapF&& map_kernel,
194  const typename std::result_of<MapF&(ForwardIt, ForwardIt, Args&&...)>::type&
195  init,
196  Args&&... args) {
197  using itr_traits = distributed_iterator_traits<ForwardIt>;
198  using mapped_t =
199  typename std::result_of<MapF&(ForwardIt, ForwardIt, Args && ...)>::type;
200  static_assert(
201  !std::is_same<mapped_t, bool>::value,
202  "distributed-map kernels returning bool are not supported (yet)");
203  using opt_mapped_t = typename optional_vector<mapped_t>::entry_t;
204 
205  auto localities = itr_traits::localities(first, last);
206  size_t i = 0;
207  rt::Handle h;
208  auto d_args = std::make_tuple(map_kernel, first, last, args...);
209  optional_vector<mapped_t> opt_res(localities.size(), init);
210  for (auto locality = localities.begin(), end = localities.end();
211  locality != end; ++locality, ++i) {
212  rt::asyncExecuteAtWithRet(
213  h, locality,
214  [](rt::Handle&, const typeof(d_args)& d_args, opt_mapped_t* result) {
215  auto first = ::std::get<1>(d_args);
216  auto last = ::std::get<2>(d_args);
217  auto lrange = itr_traits::local_range(first, last);
218  if (lrange.begin() != lrange.end()) {
219  result->valid = true;
220  result->value = apply_from<1>(
221  ::std::get<0>(d_args), ::std::forward<typeof(d_args)>(d_args));
222  } else {
223  result->valid = false;
224  }
225  },
226  d_args, &opt_res.data[i]);
227  }
228  rt::waitForCompletion(h);
229  std::vector<mapped_t> res;
230  for (auto& x : opt_res.data)
231  if (x.valid) res.push_back(x.value);
232  return res;
233 }

287 template <typename ForwardIt, typename MapF>
288 std::vector<typename std::result_of<MapF&(ForwardIt, ForwardIt)>::type>
289 local_map_init(
290  ForwardIt first, ForwardIt last, MapF&& map_kernel,
291  const typename std::result_of<MapF&(ForwardIt, ForwardIt)>::type& init) {
292  using mapped_t = typename std::result_of<MapF&(ForwardIt, ForwardIt)>::type;
293  static_assert(
294  !std::is_same<mapped_t, bool>::value,
295  "distributed-map kernels returning bool are not supported (yet)");
296 
297  // allocate partial results
298  auto parts = local_iterator_traits<ForwardIt>::partitions(
299  first, last, rt::impl::getConcurrency());
300 
301  std::vector<mapped_t> map_res(parts.size(), init);
302 
303  if (parts.size()) {
304  rt::Handle map_h;
305  size_t part_id = 0;
306  for (auto pit = parts.begin(); pit != parts.end(); ++pit) {
307  auto map_args = std::make_tuple(pit->begin(), pit->end(), map_kernel,
308  &map_res[part_id]);
309  rt::asyncExecuteAt(
310  map_h, rt::thisLocality(),
311  [](rt::Handle&, const typeof(map_args)& map_args) {
312  auto pfirst = std::get<0>(map_args);
313  auto plast = std::get<1>(map_args);
314  auto map_kernel = std::get<2>(map_args);
315  auto res_unit = std::get<3>(map_args);
316  // map over the partition
317  assert(pfirst != plast);
318  *res_unit = map_kernel(pfirst, plast);
319  },
320  map_args);
321  ++part_id;
322  }
323  rt::waitForCompletion(map_h);
324  }
325 
326  return map_res;
327 }
