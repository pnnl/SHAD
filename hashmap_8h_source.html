388 inline std::pair<
389  typename Hashmap<KTYPE, VTYPE, KEY_COMPARE, INSERT_POLICY>::iterator, bool>
390 Hashmap<KTYPE, VTYPE, KEY_COMPARE, INSERT_POLICY>::Insert(const KTYPE &key,
391  const VTYPE &value) {
392  using itr_traits = distributed_iterator_traits<iterator>;
393  size_t targetId = shad::hash<KTYPE>{}(key) % rt::numLocalities();
394  rt::Locality targetLocality(targetId);
395  std::pair<iterator, bool> res;
396 
397  if (targetLocality == rt::thisLocality()) {
398  auto lres = localMap_.Insert(key, value);
399  res.first = itr_traits::iterator_from_local(begin(), end(), lres.first);
400  res.second = lres.second;
401  } else {
402  auto insertLambda =
403  [](const std::tuple<iterator, iterator, InsertArgs> &args_,
404  std::pair<iterator, bool> *res_ptr) {
405  auto &args(std::get<2>(args_));
406  auto mapPtr = HmapT::GetPtr(args.oid);
407  auto lres = mapPtr->localMap_.Insert(args.key, args.value);
408  res_ptr->first = itr_traits::iterator_from_local(
409  std::get<0>(args_), std::get<1>(args_), lres.first);
410  res_ptr->second = lres.second;
411  };
412  rt::executeAtWithRet(
413  targetLocality, insertLambda,
414  std::make_tuple(begin(), end(), InsertArgs{oid_, key, value}), &res);
415  }
416  return res;

499  const KTYPE &key, VTYPE *res) {
500  size_t targetId = shad::hash<KTYPE>{}(key) % rt::numLocalities();
501  rt::Locality targetLocality(targetId);
502 
503  if (targetLocality == rt::thisLocality()) {
504  return localMap_.Lookup(key, res);
505  } else {
506  auto lookupLambda = [](const LookupArgs &args, LookupResult *res) {
507  auto mapPtr = HmapT::GetPtr(args.oid);
508  res->found = mapPtr->localMap_.Lookup(args.key, &res->value);
509  };
510  LookupArgs args = {oid_, key};
511  LookupResult lres;
512  rt::executeAtWithRet(targetLocality, lookupLambda, args, &lres);
513  if (lres.found) {
514  *res = std::move(lres.value);
515  }
516  return lres.found;
517  }
518  return false;
519 }
