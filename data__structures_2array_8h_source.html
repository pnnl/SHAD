693 static std::pair<rt::Locality, size_t> getTargetLocalityFromTargePosition(
694  const std::vector<std::pair<size_t, size_t>> &dataDistribution,
695  size_t position) {
696  auto itr = std::lower_bound(
697  dataDistribution.begin(), dataDistribution.end(), position,
698  [](const std::pair<size_t, size_t> &lhs, const size_t position) -> bool {
699  return lhs.second < position;
700  });
701 
702  rt::Locality dest(std::distance(dataDistribution.begin(), itr));
703  size_t off = position - itr->first;
704  return std::make_pair(dest, off);
705 }

742 void Array<T>::InsertAt(const size_t pos, const T *values,
743  const size_t numValues) {
744  size_t tgtPos = 0, firstPos = pos;
745  rt::Locality tgtLoc;
746  size_t remainingValues = numValues;
747  size_t chunkSize = 0;
748  T *valuesPtr = const_cast<T *>(values);
749 
750  while ((remainingValues > 0) &&
751  (firstPos < pivot_ * (size_ / rt::numLocalities()))) {
752  tgtLoc = rt::Locality(firstPos / (size_ / rt::numLocalities()));
753  tgtPos = firstPos % (size_ / rt::numLocalities());
754  chunkSize =
755  std::min((size_ / rt::numLocalities() - tgtPos), remainingValues);
756  if (tgtLoc == rt::thisLocality()) {
757  memcpy(&data_[tgtPos], valuesPtr, chunkSize * sizeof(T));
758  } else {
759  chunkSize = constants::min(chunkSize, kMaxChunkSize);
760  size_t argsSize =
761  sizeof(oid_) + sizeof(size_t) * 2 + sizeof(T) * chunkSize;
762  std::shared_ptr<uint8_t> args(new uint8_t[argsSize],
763  std::default_delete<uint8_t[]>());
764  uint8_t *argsPtr = args.get();
765  memcpy(argsPtr, &oid_, sizeof(oid_));
766  argsPtr += sizeof(oid_);
767  memcpy(argsPtr, &tgtPos, sizeof(size_t));
768  argsPtr += sizeof(size_t);
769  memcpy(argsPtr, &chunkSize, sizeof(size_t));
770  argsPtr += sizeof(size_t);
771  memcpy(argsPtr, valuesPtr, sizeof(T) * chunkSize);
772 
773  rt::executeAt(tgtLoc, RangedInsertAtFun, args, argsSize);
774  }
775 
776  firstPos += chunkSize;
777  remainingValues -= chunkSize;
778  valuesPtr += chunkSize;
779  }
780  while (remainingValues > 0) {
781  size_t newPos = firstPos - (pivot_ * (size_ / rt::numLocalities()));
782  tgtLoc =
783  rt::Locality(pivot_ + newPos / ((size_ / rt::numLocalities() + 1)));
784  tgtPos = newPos % ((size_ / rt::numLocalities() + 1));
785  chunkSize =
786  std::min((size_ / rt::numLocalities() + 1 - tgtPos), remainingValues);
787  if (tgtLoc == rt::thisLocality()) {
788  memcpy(&data_[tgtPos], valuesPtr, chunkSize * sizeof(T));
789  } else {
790  chunkSize = constants::min(chunkSize, kMaxChunkSize);
791  size_t argsSize =
792  sizeof(oid_) + sizeof(size_t) * 2 + sizeof(T) * chunkSize;
793  // FIXME(SHAD-125)
794  std::shared_ptr<uint8_t> args(new uint8_t[argsSize],
795  std::default_delete<uint8_t[]>());
796  uint8_t *argsPtr = args.get();
797  memcpy(argsPtr, &oid_, sizeof(oid_));
798  argsPtr += sizeof(oid_);
799  memcpy(argsPtr, &tgtPos, sizeof(size_t));
800  argsPtr += sizeof(size_t);
801  memcpy(argsPtr, &chunkSize, sizeof(size_t));
