98  void FlushBuffer() {
99  if (size_ == 0) return;
100  FlushArgs args = {data_, size_, oid_};
101  auto InsertBufferLambda = [](const FlushArgs& args) {
102  auto dsPtr = DataStructure::GetPtr(args.oid);
103  for (size_t i = 0; i < args.numEntries; i++) {
104  dsPtr->BufferEntryInsert(args.data[i]);
105  }
106  };
107  rt::executeAt(tgtLoc_, InsertBufferLambda, args);
108  size_ = 0;
109  }
