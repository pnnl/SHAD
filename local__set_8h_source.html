362 bool LocalSet<T, ELEM_COMPARE>::Find(const T& element) {
363  size_t bucketIdx = shad::hash<T>{}(element) % numBuckets_;
364  Bucket* bucket = &(buckets_array_[bucketIdx]);
365 
366  while (bucket != nullptr) {
367  for (size_t i = 0; i < bucket->BucketSize(); ++i) {
368  Entry* entry = &bucket->getEntry(i);
369  // Stop at the first empty entry.
370  if (entry->state == EMPTY) return false;
371  // Yield on pending entries.
372  while (entry->state == PENDING_INSERT) rt::impl::yield();
373  // Entry is USED.
374  if (ElemComp_(&entry->element, &element) == 0) {
375  return true;
376  }
377  }
378  bucket = bucket->next.get();
379  }
380  return false;

403 template <typename T, typename ELEM_COMPARE>
404 void LocalSet<T, ELEM_COMPARE>::Erase(const T& element) {
405  size_t bucketIdx = shad::hash<T>{}(element) % numBuckets_;
406  Bucket* bucket = &(buckets_array_[bucketIdx]);
407  Entry* prevEntry = nullptr;
408  Entry* toDelete = nullptr;
409  Entry* lastEntry = nullptr;
410  for (;;) {
411  for (size_t i = 0; i < bucket->BucketSize(); ++i) {
412  Entry* entry = &bucket->getEntry(i);
413  // 1. Key not found, returning
414  if (entry->state == EMPTY) {
415  if (toDelete != nullptr)
416  throw std::logic_error(
417  "A problem occured with"
418  "the set erase operation");
419  return;
420  }
421  while (entry->state == PENDING_INSERT) {
422  rt::impl::yield();
423  }
424  if (ElemComp_(&entry->element, &element) == 0) {
425  // 2. Key found, try to acquire a lock on it
426  if (!__sync_bool_compare_and_swap(&entry->state, USED,
427  PENDING_INSERT)) {
428  Erase(element);
429  return;
430  }
431  // 3. The entry to remove has been found,
432  // and its status set to PENDING_INSERT
433  toDelete = entry;
434  prevEntry = entry;
435  --size_;
436 
437  // now look for the last two entries.
438  size_t j = i + 1;
439  for (;;) {
440  for (; j < bucket->BucketSize(); ++j) {
441  lastEntry = &bucket->getEntry(j);
442  if (__sync_bool_compare_and_swap(&lastEntry->state, EMPTY,
443  PENDING_INSERT)) {
444  // 3. Last entry found (EMPTY->PENDING)
445  if (prevEntry == toDelete) { // just set it to EMPTY and return;
446  lastEntry->state = EMPTY;
447  toDelete->state = EMPTY;
448  return;
449  }
450  // STATUS:
451  // toDelete found and status is PENDING_INSERT
452  // lastEntry found and status is PENDING_INSERT
453  // need to find prevEntry
454 
455  if (!__sync_bool_compare_and_swap(&prevEntry->state, USED,
456  PENDING_INSERT)) {
457  rt::impl::yield();
458  lastEntry->state = EMPTY;
459  toDelete->state = USED;
460  ++size_;
461  Erase(element);
462  return;
463  }
464  // now prevEntry is locked
465  // 4. free the last entry
466  lastEntry->state = EMPTY;
467  // move prevEntry into toDelete
468  toDelete->element = std::move(prevEntry->element);
469  toDelete->state = USED;
470  // free prevEntry
471  prevEntry->state = EMPTY;
472  return;
473  } else {
474  if (lastEntry->state == PENDING_INSERT) {
475  toDelete->state = USED;
476  ++size_;
477  Erase(element);
478  return;
479  }
480  }
481  prevEntry = lastEntry;
482  }
483  j = 0;
484  if (bucket->next != nullptr) {
485  bucket = bucket->next.get();
486  } else {
487  // STATUS last entry is not empty and has not been locked
488  if (lastEntry == nullptr) {
489  // toDelete has not been found or
490  // it is the last entry at the end of the last bucket
491  if (toDelete != nullptr) toDelete->state = EMPTY;
492  return;
493  }
494  if (!__sync_bool_compare_and_swap(&lastEntry->state, USED,
495  PENDING_INSERT)) {
496  toDelete->state = USED;
497  ++size_;
498  Erase(element);
499  return;
500  }
501  if (lastEntry == prevEntry) {
502  if (toDelete == prevEntry) {
503  // No move is necessary, just set to EMPTY
504  lastEntry->state = EMPTY;
505  toDelete->state = EMPTY;
506  return;
507  } else {
508  toDelete->element = std::move(lastEntry->element);
509  toDelete->state = USED;
510  lastEntry->state = EMPTY;
511  return;
512  }
513  } else {
514  // FIXME double check if this state is reachable
515  if (toDelete == prevEntry) {
516  toDelete->element = std::move(lastEntry->element);
517  toDelete->state = USED;
518  lastEntry->state = EMPTY;
519  return;
520  } else {
521  // Need to lock prev entry as well
522  while (!__sync_bool_compare_and_swap(&prevEntry->state, USED,
523  PENDING_INSERT)) {
524  rt::impl::yield();
525  }
526  lastEntry->state = EMPTY;
527  toDelete->element = std::move(prevEntry->element);
528  toDelete->state = USED;
529  prevEntry->state = EMPTY;
530  }
531  }
532  return;
533  }
534  }
535  }
536  prevEntry = entry;
537  }
538  if (bucket->next != nullptr) {
539  bucket = bucket->next.get();
540  } else {
541  return;
542  }
543  }
544 }

557 template <typename T, typename ELEM_COMPARE>
558 std::pair<typename LocalSet<T, ELEM_COMPARE>::iterator, bool>
559 LocalSet<T, ELEM_COMPARE>::Insert(const T& element) {
560  size_t bucketIdx = shad::hash<T>{}(element) % numBuckets_;
561  Bucket* bucket = &(buckets_array_[bucketIdx]);
562 
563  // Forever or until we find an insertion point.
564  for (;;) {
565  for (size_t i = 0; i < bucket->BucketSize(); ++i) {
566  Entry* entry = &bucket->getEntry(i);
567 
568  if (__sync_bool_compare_and_swap(&entry->state, EMPTY, PENDING_INSERT)) {
569  entry->element = std::move(element);
570  ++size_;
571  entry->state = USED;
572  return std::make_pair(iterator(this, bucketIdx, i, bucket, entry),
573  true);
574  } else {
575  while (entry->state == PENDING_INSERT) rt::impl::yield();
576  if (ElemComp_(&entry->element, &element) == 0) {
577  return std::make_pair(iterator(this, bucketIdx, i, bucket, entry),
578  false);
579  }
580  }
581  }
582 
583  if (bucket->next == nullptr) {
584  // We need to allocate a new buffer
585  if (__sync_bool_compare_and_swap(&bucket->isNextAllocated, false, true)) {
586  // Allocate the bucket
587  std::shared_ptr<Bucket> newBucket(
588  new Bucket(constants::kSetDefaultNumEntriesPerBucket));
589  bucket->next.swap(newBucket);
590  } else {
591  // Wait for the allocation to happen
592  while (bucket->next == nullptr) rt::impl::yield();
593  }
594  }
595 
596  bucket = bucket->next.get();
597  }
598 }
